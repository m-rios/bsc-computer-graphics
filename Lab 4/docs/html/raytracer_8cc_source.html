69 void Raytracer::raytrace(double origin[3], double direction[3], double rgb[3],double contribution) {
70  int i;
71  double closestDistance, distance;
72  Object *closestObject;
73  set<Object*>::iterator objIterator;
74  set<Object*>::iterator objIteratorEnd;
75  set<Light*>::iterator lightIterator;
76  set<Light*>::iterator lightIteratorEnd;
77 
78  if(debugThisPixel) {
79  printDebugIndentation(); debugIndentation++; printf("-> Raytrace\n");
80  }
81 
82  /* Iterate over all objects and find the closest one that
83  intersects this ray. */
84  for(closestDistance=MAX_DISTANCE,closestObject=NULL,
85  objIterator=objects->begin(),objIteratorEnd=objects->end();
86  objIterator != objIteratorEnd;objIterator++) {
87  Object *object = *objIterator;
88  distance = object->lineTest(origin,direction,closestDistance);
89  if(distance < closestDistance && distance > 1e-5) {
90  closestDistance = distance;
91  closestObject = object;
92  }
93  }
94 
95  if(closestDistance >= MAX_DISTANCE) {
96  /* No objects hit, assign background colour to ray instead. */
97  for(i=0;i<3;i++) rgb[i] = background[i];
98  if(debugThisPixel) { debugIndentation--; printDebugIndentation(); printf("<- Miss\n"); }
99  return;
100  }
101 
102  double point[3], normal[3];
103  for(i=0;i<3;i++) point[i] = origin[i]+closestDistance*direction[i];
104  closestObject->getNormal(point,normal);
105 
106  /* Get lighting properties for this point */
107  LightingProperties properties;
108  closestObject->getLightingProperties(point,&properties,normal);
109  normalize(normal);
110 
111  double E[3]; /* Vector towards eye. */
112  for(i=0;i<3;i++) E[i] = -direction[i];
113  normalize(E);
114 
115  /* Now, compute the colour for this point */
116  /* Ambient light first */
117  for(i=0;i<3;i++) rgb[i] = properties.ambient[i] * ambientLight[i];
118 
119  /* Iterate over all the lights and add their colours to the rgb
120  colour using the Blinn-Phong shading model. */
121  for(lightIterator=lights->begin(),lightIteratorEnd=lights->end();
122  lightIterator!=lightIteratorEnd;lightIterator++) {
123  Light *light = *lightIterator;
124  double L[3]; /* Light vector */
125  sub(light->position,point,L);
126  double lightDistance=length(L);
127  for(i=0;i<3;i++) L[i]=L[i]/lightDistance;
128  /* First, cast a shadow feeler */
129  for(objIterator=objects->begin(),objIteratorEnd=objects->end();
130  objIterator != objIteratorEnd;objIterator++) {
131  Object *object = *objIterator;
132  if(object == closestObject)
133  /* For now, ignore shadows cast on ourselves */
134  continue;
135  distance = object->lineTest(point,L,lightDistance);
136  if(distance < lightDistance) break;
137  }
138  if(objIterator != objIteratorEnd)
139  /* A shadow was found, so ignore this light */
140  continue;
141 
142  /*printf("L: %3.1f %3.1f %3.1f\n",L[0],L[1],L[2]);*/
143  double diffusePower = dotProduct(normal,L);
144  if(diffusePower > 0) {
145  /* Light is shining on the front of the object. */
146  for(i=0;i<3;i++) rgb[i] += diffusePower * light->colour[i] * properties.diffuse[i];
147 
148  double RL[3]; /* Reflection of light vector */
149  double LdotN = dotProduct(L,normal);
150  for(i=0;i<3;i++) RL[i] = 2.0 * LdotN * normal[i] - L[i];
151 
152  double specDot=dotProduct(RL,E);
153  if(specDot > 0.0) {
154  double specularPower = pow(specDot, properties.shininess);
155  for(i=0;i<3;i++) {
156  rgb[i] += specularPower * properties.specular[i];
157  }
158  }
159  } else {
160  /* Light is shining from behind object... ignore for now. To be
161  slightly more correct we should also allow the light to shine
162  through the object and multiply it by the transparent colour
163  of the object. */
164  }
165  }
166 
167  /* Add light that is reflected on object */
168  double reflection = 0.4*properties.reflection[0]+0.4*properties.reflection[1]+0.2*properties.reflection[2];
169  if(contribution*reflection > 0.05) {
170  double rgbTmp[3]; /* Temporary colour for incoming light */
171  double R[3]; /* Reflection vector */
172  double s=2.0*dotProduct(E,normal);
173  double pointR[3]; /* From from where to trace */
174  for(i=0;i<3;i++) {
175  R[i]=s*normal[i] - E[i];
176  pointR[i]=point[i]+R[i]*1e-4; /* To void numberical instability */
177  }
178  /* Recurse on this ray to get incoming light level */
179  raytrace(point,R,rgbTmp,contribution*reflection);
180  /* Add the incomming light to the colour of this pixel */
181  for(i=0;i<3;i++) rgb[i] += rgbTmp[i]*properties.reflection[i];
182  }
183 
184  if(debugThisPixel) {
185  debugIndentation--; printDebugIndentation(); printf("<- RGB %.1f %.1f %.1f\n",rgb[0],rgb[1],rgb[2]);
186  }
187 }
