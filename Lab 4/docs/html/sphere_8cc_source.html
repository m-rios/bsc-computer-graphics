32 double Sphere::lineTest(double O[3],double dir[3],double maxDistance) {
33  /* Solving this lineIntersection test is equal to solving the second
34  degree formula "a X^2 + b X + C = 0" for a, b, c given below. */
35  //printf("ray: (%3.2f %3.2f %3.2f) + alpha * (%3.2f %3.2f %3.2f)\n",O[0],O[1],O[2],dir[0],dir[1],dir[2]);
36  double a = dotProduct(dir,dir);
37  double b = 2 * dotProduct(O,dir);
38  double c = dotProduct(O,O) - radius*radius;
39  double s = b * b - 4*a*c;
40  if(s < 0) { return MAX_DISTANCE; }
41  s = sqrt(s);
42  double sol1 = (-b - s)/(2*a);
43  if(sol1 > 0) return sol1;
44  double sol2 = (-b + s)/(2*a);
45  if(sol2 > 0) return sol2;
46  return MAX_DISTANCE;
47 }
