38 void Camera::setFocus(double focus[3]) {
39  int i;
40  double dot;
41  sub(focus,origin,forward);
42  normalize(forward);
43  dot=dotProduct(forward,up);
44  if(dot == 1.0) {
45  printf("Error - invalid use of setFocus. Focal point coincides with up vector.\n");
46  exit(0);
47  }
48  /*printf("dot = %3.1f\n",dot);*/
49  for(i=0;i<3;i++) up[i] -= dot*forward[i];
