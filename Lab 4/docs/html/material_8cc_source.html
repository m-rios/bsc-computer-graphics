47 void NoiseMaterial::getLightingProperties(double point[3],LightingProperties *props,double normal[3]) {
48  double scale=1.0;
49  double amplitude=1.0;
50  double offset[3];
51  int i;
52  double v=0.0;
53  zero(offset);
54  for(i=0;i<6;i++) {
55  v+=amplitude*noise(point[0]*scale+offset[0],point[1]*scale+offset[1],point[2]*scale+offset[2]);
56  scale *= 2.0;
57  amplitude *= 0.5;
58  }
59  v=0.5+0.3*v;
60

64  for(i=0;i<3;i++) props->diffuse[i]=v;

76 void MaterialMap::add(double nodePosition,LightingProperties*nodeProperty) {
77  int insertionPoint;
78  int i;
79  /* First, find correct place in list of nodes to insert this node in */
80  for(insertionPoint=0;insertionPoint<nNodes;insertionPoint++)
81  if(nodePositions[insertionPoint] > nodePosition) break;
82  /* Move all other existing nodes forward one step to make place
83  for this one */
84  for(i=nNodes;i>insertionPoint;i--) {
85  nodePositions[i]=nodePositions[i-1];
86  nodeProperties[i]=nodeProperties[i-1];
87  }
88  nNodes++;
89  /* Insert this node into the proper place */
90  nodePositions[insertionPoint]=nodePosition;
91  nodeProperties[insertionPoint]=*nodeProperty;
92 }
93 void MaterialMap::getLightingProperties(double point[3],LightingProperties *props,double normal[3]) {
94  int i;
95  double val;
96 
97  double scale=1.0;
98  double amplitude=1.0;
99  double offset[3];
100  val=0.0;
101  zero(offset);
102  for(i=0;i<6;i++) {
103  val+=amplitude*noise(point[0]*scale+offset[0],point[1]*scale+offset[1],point[2]*scale+offset[2]);
104  scale *= 2.0;
105  amplitude *= 0.5;
106  }
107 
108  int highNode;
109  int lowNode;
110  double fracL;
111  double fracH;
112  /* TODO - do this search in a smarter way, eg. start at nNodes/2 and
113  with a step size of nNodes/4 etc. */
114  for(highNode=0,lowNode=0;highNode<nNodes;highNode++,lowNode=highNode-1)
115  if(nodePositions[highNode] > val) break;
116  if(highNode == nNodes) { *props = nodeProperties[nNodes-1]; }
117  else if(lowNode == highNode) { *props = nodeProperties[0]; }
118  else {
119  double delta=nodePositions[highNode]-nodePositions[lowNode];
120  if(delta < 1e-5) *props = nodeProperties[lowNode];
121  else {
122  fracH = (val - nodePositions[lowNode])/delta;
123  fracL = 1.0 - fracH;
124  LightingProperties *low=&nodeProperties[lowNode],*high=&nodeProperties[highNode];
125  props->ambient[0] = fracH*high->ambient[0] + fracL*low->ambient[0];
126  props->ambient[1] = fracH*high->ambient[1] + fracL*low->ambient[1];
