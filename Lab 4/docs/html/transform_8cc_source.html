121 void Transform::computeInverseTransform() {
122  /* Use gauss-jordan's algorithm to invert the forward matrix to create the inverse matrix */
123  int i,j,k;
124  double M[4][8]; /* Will hold the composite [ forward | identity ]
125  and be solved */
126 
127  for(i=0;i<4;i++)
128  for(j=0;j<4;j++) {
129  M[i][j]=forward[i][j];
130  M[i][j+4] = (i==j?1.0:0.0);
131  }
132  /* TODO - add pivot operations to yield better nummerical stability */
133  /* Step 1: eliminate M[i][i] from M[i+1][..] and down */
134  for(i=0;i<3;i++) {
135  for(j=i+1;j<4;j++) {
136  for(k=i+1;k<8;k++)
137  M[j][k] -= M[j][i]/M[i][i] * M[i][k];
138  M[j][i]=0.0;
139  }
140  }
141  /* Step 2: substitute upwards */
142  for(i=3;i>=0;i--) {
143  for(k=i+1;k<8;k++)
144  M[i][k] /= M[i][i];
145  M[i][i] = 1.0;
146  for(j=0;j<i;j++) {
147  for(k=i+1;k<8;k++)
148  M[j][k] -= M[i][k] * M[j][i];
149  M[j][i] = 0.0;
150  }
151  }
152  /* The result is now in M[4..7][0..3] */
153  for(i=0;i<4;i++)
154  for(j=0;j<4;j++) {
155  inverse[i][j] = M[i][j+4];
156  }
157 }
